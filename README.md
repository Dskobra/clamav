build-essential, 
openssl, 
libssl-dev, 
libcurl4-openssl-dev, 
zlib1g-dev, 
libpng-dev, 
libxml2-dev, 
libjson-c-dev, 
libbz2-dev, 
libpcre3-dev, 
libncurses-dev, 
libmilter1.0.1, 
libmilter-dev, 
valgrind, 
check
